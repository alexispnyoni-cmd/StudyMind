SCHOOLS TABLE ──────────────────────────────── create table if not exists public.schools ( id uuid primary key default gen_random_uuid(), name text not null, slug text unique not null, -- e.g. "greenwood-high" address text, province text, country text default 'South Africa', logo_url text, primary_color text default '#7c6ff7', created_at timestamptz default now() ); alter table public.schools enable row level security; create policy "Anyone can view schools" on public.schools for select using (true); create policy "Admins can update their school" on public.schools for update using ( exists ( select 1 from public.profiles where id = auth.uid() and school_id = schools.id and role = 'admin' ) ); -- ─── 2. UPDATE PROFILES TABLE ──────────────────────── -- Add school_id if it doesn't exist alter table public.profiles add column if not exists school_id uuid references public.schools(id), add column if not exists subject text, add column if not exists grade text, add column if not exists avatar_emoji text default '🎒', add column if not exists is_active boolean default true, add column if not exists updated_at timestamptz default now(); -- Migrate existing school_name data to school_id where possible update public.profiles p set school_id = '00000000-0000-0000-0000-000000000001' where p.school_id is null and exists ( select 1 from public.schools s where s.name = p.school_name ); -- Drop old RLS policies and recreate properly drop policy if exists "Users can view all profiles" on public.profiles; drop policy if exists "Users can update own profile" on public.profiles; drop policy if exists "Users can insert own profile" on public.profiles; create policy "Users can view profiles in same school" on public.profiles for select using ( auth.uid() = id or school_id = (select school_id from public.profiles where id = auth.uid()) ); create policy "Users can update own profile" on public.profiles for update using (auth.uid() = id); create policy "Users can insert own profile" on public.profiles for insert with check (auth.uid() = id); -- ─── 4. UPDATE TRIGGER — include school_id ─────────── create or replace function public.handle_new_user() returns trigger as $$ declare v_school_id uuid; v_school_name text; begin -- Get school_name from metadata v_school_name := coalesce(new.raw_user_meta_data->>'school_name', 'Greenwood High School'); -- Look up school_id by name select id into v_school_id from public.schools where name ilike v_school_name limit 1; -- If not found, use default school if v_school_id is null then v_school_id := '00000000-0000-0000-0000-000000000001'; end if; insert into public.profiles ( id, full_name, role, school_id, school_name, grade, subject, avatar_emoji ) values ( new.id, coalesce(new.raw_user_meta_data->>'full_name', split_part(new.email, '@', 1)), coalesce(new.raw_user_meta_data->>'role', 'student'), v_school_id, v_school_name, coalesce(new.raw_user_meta_data->>'grade', ''), coalesce(new.raw_user_meta_data->>'subject', ''), coalesce(new.raw_user_meta_data->>'avatar_emoji', '🎒') ) on conflict (id) do update set school_id = excluded.school_id, school_name = excluded.school_name, updated_at = now(); return new; end; $$ language plpgsql security definer; drop trigger if exists on_auth_user_created on auth.users; create trigger on_auth_user_created after insert on auth.users for each row execute procedure public.handle_new_user(); -- ─── 5. USEFUL VIEWS ──────────────────────────────── -- Student overview (for parent/admin dashboards) create or replace view public.student_overview as select p.id, p.full_name, p.grade, p.school_id, p.school_name, count(distinct h.id) filter (where h.done = false) as pending_homework, count(distinct e.id) filter (where e.exam_date >= current_date) as upcoming_exams, count(distinct a.id) filter (where a.attendance_date >= current_date - 30 and a.status = 'present')::numeric / nullif(count(distinct a.id) filter (where a.attendance_date >= current_date - 30), 0) * 100 as attendance_pct from public.profiles p left join public.homework h on h.student_id = p.id left join public.exams e on e.student_id = p.id left join public.attendance a on a.student_id = p.id where p.role = 'student' group by p.id, p.full_name, p.grade, p.school_id, p.school_name; -- ═══════════════════════════════════════════════════════ -- DONE — Phase 1 schema is ready. -- ═══════════════════════════════════════════════════════
